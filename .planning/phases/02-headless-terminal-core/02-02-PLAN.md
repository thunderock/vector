   use crate::listener::NoopListener;
   use crate::parser::Processor;

   pub struct Term {
       inner: AlacrittyTerm<NoopListener>,
       parser: Processor,
       cols: u16,
       rows: u16,
   }

   impl Term {
       pub fn new(cols: u16, rows: u16, scrollback: usize) -> Self {
           let mut config = Config::default();
           config.{ACTUAL_SCROLLBACK_FIELD_FROM_SPIKE} = scrollback;
           let dims = VectorDims { cols: cols.into(), rows: rows.into() };
           let inner = AlacrittyTerm::new(config, &dims, NoopListener);
           let parser = Processor::new();
           Self { inner, parser, cols, rows }
       }

       pub fn feed(&mut self, bytes: &[u8]) {
           // Pitfall 4: feed bytes directly. Never decode UTF-8 here.
           self.parser.advance(&mut self.inner, bytes);
       }

       pub fn resize(&mut self, cols: u16, rows: u16) {
           let dims = VectorDims { cols: cols.into(), rows: rows.into() };
           self.inner.resize(dims);
           self.cols = cols;
           self.rows = rows;
       }

       pub fn grid(&self) -> &alacritty_terminal::grid::Grid<alacritty_terminal::term::cell::Cell> {
           self.inner.grid()
       }

       pub fn cursor(&self) -> (u16, u16) {
           let c = self.inner.grid().cursor.point;
           (c.column.0 as u16, c.line.0 as u16)
       }

       pub fn mode(&self) -> TermMode {
           self.inner.mode()
       }

       pub fn dims(&self) -> (u16, u16) { (self.cols, self.rows) }

       pub(crate) fn inner(&self) -> &AlacrittyTerm<NoopListener> { &self.inner }
   }

   //! Library-only regex search across visible grid + scrollback (CORE-03, D-39).

   use alacritty_terminal::term::search::{RegexSearch, RegexIter};
   use alacritty_terminal::{ACTUAL_POINT_PATH}::{Point, Direction};
   use regex::Regex;

   use crate::term::Term;

   #[derive(Debug, Clone, Copy, PartialEq, Eq)]
   pub struct Match {
       pub start_row: i32,
       pub start_col: u16,
       pub end_row: i32,
       pub end_col: u16,
   }

   impl Term {
       pub fn search(&self, regex: &Regex) -> Vec<Match> {
           let mut dfa = match RegexSearch::new(regex.as_str()) {
               Ok(d) => d,
               Err(_) => return Vec::new(),
           };
           let topmost = self.inner().{ACTUAL_TOPMOST_LINE_METHOD}();
           let bottommost = self.inner().{ACTUAL_BOTTOMMOST_LINE_METHOD}();
           let cols = self.inner().{ACTUAL_COLUMNS_METHOD}();
           let start = Point::new(topmost, 0.into());
           let end = Point::new(bottommost, (cols - 1).into());
           RegexIter::new(start, end, Direction::Right, self.inner(), &mut dfa)
               .map(|m| Match {
                   start_row: m.start().line.0 as i32,
                   start_col: m.start().column.0 as u16,
                   end_row: m.end().line.0 as i32,
                   end_col: m.end().column.0 as u16,
               })
               .collect()
       }
   }
