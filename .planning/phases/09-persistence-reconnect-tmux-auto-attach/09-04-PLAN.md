       ```rust
       pub fn format_reconnect_text(profile: &str, attempt: u32, content_cells: u16) -> Option<String> {
           if content_cells < 18 { return None; }
           let attempt_text = if attempt >= 10 { String::from("9+") } else { attempt.to_string() };
           let suffix = format!(" (attempt {attempt_text})");
           if content_cells < 28 {
               return Some(format!("Reconnecting…{suffix}"));
           }
           let profile_render = if content_cells < 40 && profile.chars().count() > 16 {
               let chars: Vec<char> = profile.chars().collect();
               let first: String = chars.iter().take(8).collect();
               let last: String = chars.iter().rev().take(8).collect::<Vec<_>>().into_iter().rev().collect();
               format!("{first}…{last}")
           } else {
               profile.to_string()
           };
           Some(format!("Reconnecting to {profile_render}…{suffix}"))
       }
       ```
