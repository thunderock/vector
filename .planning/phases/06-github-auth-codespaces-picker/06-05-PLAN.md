   ```rust
   //! Phase 6 / AUTH-01 — tokio task driver for the device-flow state machine.
   //! Bridges to the main thread via EventLoopProxy<UserEvent>.

   use std::sync::Arc;
   use std::sync::atomic::{AtomicBool, Ordering};
   use std::time::{Duration, SystemTime, UNIX_EPOCH};

   use tokio::runtime::Handle;
   use winit::event_loop::EventLoopProxy;

   use vector_codespaces::{AuthError, GitHubAuth, TokenStore};

   use crate::UserEvent;

   /// Returned by `spawn_device_flow` so the modal's Cancel button can abort.
   #[derive(Clone)]
   pub struct AuthCancellation {
       flag: Arc<AtomicBool>,
   }

   impl AuthCancellation {
       pub fn cancel(&self) {
           self.flag.store(true, Ordering::SeqCst);
       }
       pub fn is_cancelled(&self) -> bool {
           self.flag.load(Ordering::SeqCst)
       }
   }

   pub fn spawn_device_flow(
       handle: &Handle,
       proxy: EventLoopProxy<UserEvent>,
   ) -> AuthCancellation {
       let cancel = AuthCancellation { flag: Arc::new(AtomicBool::new(false)) };
       let cancel_clone = cancel.clone();
       handle.spawn(async move {
           run_flow(proxy, cancel_clone).await;
       });
       cancel
   }

   async fn run_flow(proxy: EventLoopProxy<UserEvent>, cancel: AuthCancellation) {
       let auth = match GitHubAuth::new() {
           Ok(a) => a,
           Err(e) => {
               let _ = proxy.send_event(UserEvent::AuthFailed {
                   reason: format!("init: {e}"),
               });
               return;
           }
       };

       let (display, details) = match auth.request_device_code().await {
           Ok(v) => v,
           Err(e) => {
               let _ = proxy.send_event(UserEvent::AuthFailed {
                   reason: format!("device_code: {e}"),
               });
               return;
           }
       };

       let expires_at = SystemTime::now()
           + display
               .expires_at
               .saturating_duration_since(std::time::Instant::now());
       let _ = proxy.send_event(UserEvent::AuthDisplayCode {
           user_code: display.user_code.clone(),
           verification_uri: display.verification_uri.clone(),
           expires_at,
           interval_secs: display.interval_secs,
       });

       // Race oauth poll vs cancellation flag.
       let poll_fut = auth.poll_for_token(details);
       tokio::pin!(poll_fut);
       let tokens = loop {
           tokio::select! {
               result = &mut poll_fut => break result,
               _ = tokio::time::sleep(Duration::from_millis(200)) => {
                   if cancel.is_cancelled() {
                       let _ = proxy.send_event(UserEvent::AuthFailed {
                           reason: "cancelled".into(),
                       });
                       return;
                   }
               }
           }
       };
       let tokens = match tokens {
           Ok(t) => t,
           Err(AuthError::Expired) => {
               let _ = proxy.send_event(UserEvent::AuthFailed {
                   reason: "expired".into(),
               });
               return;
           }
           Err(AuthError::Cancelled) => {
               let _ = proxy.send_event(UserEvent::AuthFailed {
                   reason: "cancelled".into(),
               });
               return;
           }
           Err(e) => {
               let _ = proxy.send_event(UserEvent::AuthFailed {
                   reason: format!("oauth: {e}"),
               });
               return;
           }
       };

       // Persist tokens.
       let store = TokenStore::new();
       if let Err(e) = store.save_access(&tokens.access) {
           let _ = proxy.send_event(UserEvent::AuthFailed {
               reason: format!("keychain: {e}"),
           });
           return;
       }
       if let Some(refresh) = &tokens.refresh {
           let _ = store.save_refresh(refresh);
       }

       // Fetch @login (best-effort).
       let login = fetch_login(&tokens.access).await.unwrap_or_else(|_| "unknown".to_string());
       let _ = proxy.send_event(UserEvent::AuthCompleted { user_login: login });
       let _ = UNIX_EPOCH; // keep import used if SystemTime path changes
   }

   async fn fetch_login(
       token: &zeroize::Zeroizing<String>,
   ) -> Result<String, octocrab::Error> {
       let octo = octocrab::Octocrab::builder()
           .personal_token((**token).clone())
           .add_header(http::header::USER_AGENT, "Vector/0.1".into())
           .build()?;
       let user: octocrab::models::Author = octo.current().user().await?;
       Ok(user.login)
   }
   ```

   ```rust
               "Code copied to clipboard. Expires in {min:02}:{sec:02}."
           );
           unsafe {
               self.countdown_field.setStringValue(&NSString::from_str(&text));
           }
           false
       }

       /// Re-copy code and open verification URL. UI-SPEC §5.1 primary action.
       pub fn primary_action(&self, mtm: MainThreadMarker) {
           write_clipboard(mtm, &self.user_code);
           open_url(&self.verification_uri);
       }

       /// Cancel: signal actor + restore clipboard + dismiss.
       pub fn cancel(&self, mtm: MainThreadMarker) {
           self.cancellation.cancel();
           self.dismiss(mtm);
       }

       pub fn dismiss(&self, mtm: MainThreadMarker) {
           if let Some(prev) = &self.saved_clipboard {
               write_clipboard(mtm, prev);
           } else {
               clear_clipboard(mtm);
           }
           unsafe { self.panel.orderOut(None) };
       }
   }

   fn make_label(
       mtm: MainThreadMarker,
       text: &str,
       frame: NSRect,
       font_size: f64,
       muted: bool,
   ) -> Retained<NSTextField> {
       unsafe {
           let f = NSTextField::labelWithString(&NSString::from_str(text), mtm);
           f.setFrame(frame);
           f.setBezeled(false);
           f.setEditable(false);
           f.setBackgroundColor(Some(&NSColor::clearColor()));
           f.setAlignment(NSTextAlignment::Center);
           let font = NSFont::systemFontOfSize(font_size);
           f.setFont(Some(&font));
           if muted {
               f.setTextColor(Some(&NSColor::secondaryLabelColor()));
           }
           f
       }
   }

   fn make_code_field(mtm: MainThreadMarker, code: &str, frame: NSRect) -> Retained<NSTextField> {
       unsafe {
           let f = NSTextField::labelWithString(&NSString::from_str(code), mtm);
           f.setFrame(frame);
           f.setBezeled(true);
           f.setEditable(false);
           f.setSelectable(true);
           f.setAlignment(NSTextAlignment::Center);
           let font = NSFont::monospacedSystemFontOfSize_weight(32.0, 0.6_f64);
           f.setFont(Some(&font));
           f
       }
   }

   fn make_button(mtm: MainThreadMarker, title: &str, frame: NSRect) -> Retained<NSButton> {
       unsafe {
           let b = NSButton::buttonWithTitle_target_action(
               &NSString::from_str(title),
               None,
               None,
               mtm,
           );
           b.setFrame(frame);
           b.setBezelStyle(NSBezelStyle::Rounded);
           b
       }
   }

   fn capture_clipboard(_mtm: MainThreadMarker) -> Option<String> {
       unsafe {
           let pb = NSPasteboard::generalPasteboard();
           pb.stringForType(NSPasteboardTypeString).map(|s| s.to_string())
       }
   }

   fn write_clipboard(_mtm: MainThreadMarker, value: &str) {
       unsafe {
           let pb = NSPasteboard::generalPasteboard();
           pb.clearContents();
           pb.setString_forType(&NSString::from_str(value), NSPasteboardTypeString);
       }
   }

   fn clear_clipboard(_mtm: MainThreadMarker) {
       unsafe {
           let pb = NSPasteboard::generalPasteboard();
           pb.clearContents();
       }
   }

   fn open_url(url: &str) {
       unsafe {
           let ns_url = NSURL::URLWithString(&NSString::from_str(url));
           if let Some(u) = ns_url {
               let workspace = NSWorkspace::sharedWorkspace();
               let _ = workspace.openURL(&u);
           }
       }
   }
   ```
