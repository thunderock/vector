       """HARDEN-03 D-22 — Compare cargo-geiger output against the allowlist.

       Usage: check-geiger.py <allowlist.json> <geiger-output.json>

       def main():

       //! HARDEN-03 D-11 runtime gate — exercise GitHub OAuth code paths with
       //! tracing captured to an in-memory buffer; assert zero token-shaped matches.

       use std::sync::{Arc, Mutex};
       use std::io::Write;
       use regex::Regex;
       use tracing_subscriber::fmt::MakeWriter;

       #[derive(Clone)]
       struct StringWriter(Arc<Mutex<Vec<u8>>>);

       impl Write for StringWriter {
           fn write(&mut self, buf: &[u8]) -> std::io::Result<usize> {
               self.0.lock().unwrap().extend_from_slice(buf);
               Ok(buf.len())
           }
           fn flush(&mut self) -> std::io::Result<()> { Ok(()) }
       }
       impl<'a> MakeWriter<'a> for StringWriter {
           type Writer = StringWriter;
           fn make_writer(&'a self) -> Self::Writer { self.clone() }
       }

       #[tokio::test(flavor = "current_thread")]
       async fn github_oauth_does_not_leak_tokens() {
           let buf = Arc::new(Mutex::new(Vec::<u8>::new()));
           let writer = StringWriter(buf.clone());
           let _guard = tracing_subscriber::fmt()
               .with_writer(writer)
               .with_max_level(tracing::Level::DEBUG)
               .with_test_writer()
               .set_default();

           // Exercise the auth pipeline with realistic-looking but fake tokens.
           // The token MUST be regex-matchable so the gate is self-verifying.
           let fake_token = "gho_FAKE_TOKEN_FOR_TESTING_1234567890abcdef";
           // Invoke whatever functions cause tracing emission — read existing
           // wiremock-backed tests in this crate for the right entry points,
           // e.g. construct a TokenStore + call `set_tokens(fake_token, None)`,
           // OR call `CodespacesClient` with the fake bearer and let it 401 against wiremock.
           //
           // The minimum surface is: instantiate a `Tokens` and log it via Debug.
           let tokens = vector_codespaces::auth::Tokens {
               access: zeroize::Zeroizing::new(fake_token.to_string()),
               refresh: None,
           };
           tracing::debug!(?tokens, "constructed tokens — should redact");

           // Drain buffer.
           drop(_guard);
           let output = String::from_utf8(buf.lock().unwrap().clone()).unwrap_or_default();
           let re = Regex::new(r"(gho_|ghp_|gha_|ghs_|eyJ[A-Za-z0-9_-]{10,})").unwrap();
           assert!(
               !re.is_match(&output),
               "Token-shaped string leaked into tracing output: {}",
               output
           );
       }
