       ```rust
       //! HARDEN-02 perf gate probe — D-10 + D-23.
       //!
       //! Two measurements:
       //!   - idle_cpu_pct: 5s of zero PTY activity; sample own RSS+CPU via libc rusage delta.
       //!   - paste_render_fps: feed 1 MB of pre-canned ANSI into Term; render until done;
       //!     report frames/sec achieved (vsync cap 60 → expect ≥ 55 on a quiet runner).
       //!
       //! Emits a one-line JSON to stdout: {"idle_cpu_pct": X.X, "paste_render_fps": Y.Y}.
       //! Caller (CI step or shell) reads the JSON and applies the threshold.

       use std::io::Write;
       use std::time::{Duration, Instant};

       fn main() {
           let idle = measure_idle_cpu();
           let fps = measure_paste_render_fps();
           let json = format!(
               "{{\"idle_cpu_pct\":{:.3},\"paste_render_fps\":{:.2}}}",
               idle, fps
           );
           println!("{json}");
           // Also write to a side file for CI artifact upload.
           if let Ok(mut f) = std::fs::File::create("target/perf-probe.json") {
               let _ = writeln!(f, "{json}");
           }
       }

       fn measure_idle_cpu() -> f64 {
           // Build compositor, feed nothing, sample our own CPU time over 5 seconds.
           // Using std::time + rusage delta — no `ps`, no `time`, deterministic.
           let start = Instant::now();
           let start_cpu = self_cpu_seconds();
           // No-op: spin until 5 seconds wall time elapsed without doing any render work.
           while start.elapsed() < Duration::from_secs(5) {
               std::thread::sleep(Duration::from_millis(100));
           }
           let end_cpu = self_cpu_seconds();
           let cpu_used = end_cpu - start_cpu;
           let wall = start.elapsed().as_secs_f64();
           (cpu_used / wall) * 100.0
       }

       fn measure_paste_render_fps() -> f64 {
           // Build offscreen Term + Compositor; feed 1 MB of pre-canned text;
           // render N frames; report frames/wall_seconds.
           // If no Metal adapter (CI Linux fallback?), emit -1.0.
           // ... (implementation reuses vector_render::tests::common via [[example]]
           //      and a [dev-dependencies] dep on vector-render's own tests/common
           //      OR copies the ~20 lines of build_compositor verbatim.)
           let ctx = match vector_render::RenderContext::new_offscreen(800, 480) {
               Ok(c) => c, Err(_) => return -1.0,
           };
           let font_stack = match vector_fonts::FontStack::load_bundled(1.0, 14.0) {
               Ok(f) => f, Err(_) => return -1.0,
           };
           let mut comp = vector_render::Compositor::new_with(
               &ctx.device, &ctx.queue, ctx.format, ctx.width, ctx.height, font_stack,
           ).expect("compositor");
           let mut term = vector_term::Term::new(80, 24, 100_000);
           let blob = "lorem ipsum dolor sit amet, consectetur adipiscing elit\r\n".repeat(20_000); // ~1.1 MB
           term.feed(blob.as_bytes());
           let n = 60;
           let start = Instant::now();
           for _ in 0..n {
               let _ = comp.render_offscreen_with(
                   &ctx.device, &ctx.queue, ctx.width, ctx.height, &mut term, None,
               );
           }
           let elapsed = start.elapsed().as_secs_f64();
           n as f64 / elapsed
       }

       #[cfg(target_os = "macos")]
       fn self_cpu_seconds() -> f64 {
           // libc::getrusage(RUSAGE_SELF) returns user+sys CPU time.
           unsafe {
               let mut ru: libc::rusage = std::mem::zeroed();
               libc::getrusage(libc::RUSAGE_SELF, &mut ru);
               let utime = ru.ru_utime.tv_sec as f64 + ru.ru_utime.tv_usec as f64 / 1_000_000.0;
               let stime = ru.ru_stime.tv_sec as f64 + ru.ru_stime.tv_usec as f64 / 1_000_000.0;
               utime + stime
           }
       }
       #[cfg(not(target_os = "macos"))]
       fn self_cpu_seconds() -> f64 { 0.0 }
       ```
