    ```rust
    //! POLISH-07 / D-75 / UI-SPEC §5.1 — title-bar tint stripe.
    //!
    //! Dedicated wgpu render pass: 1 pipeline, 1 quad, 1 solid-color uniform.
    //! Reuses the cell pipeline's vertex format (Plan 03-03) but binds a single-color
    //! fragment shader. Sits OVER the top 28 px of the window content area, BELOW
    //! the AppKit titlebar.

    use wgpu::{
        BindGroup, BindGroupLayout, Buffer, Device, RenderPass, RenderPipeline,
    };

    pub struct TintStripePipeline {
        pipeline: RenderPipeline,
        // bind groups for the color uniform; populated on construction
        bg_layout: BindGroupLayout,
        quad_vertices: Buffer,
        // Current color uniform — None → don't render (collapse to 0 px).
        color_uniform: Buffer,
        current_color: Option<[f32; 4]>,
    }

    impl TintStripePipeline {
        pub fn new(device: &Device, surface_format: wgpu::TextureFormat) -> Self {
            // 1. compile WGSL: vertex pass-through + fragment that samples a single color uniform.
            // 2. create pipeline layout with 1 bind group (color uniform).
            // 3. allocate the quad vertex buffer with 6 vertices spanning [-1, 1] × [top_y, top_y - h_ndc]
            //    where top_y = 1.0 and h_ndc = 28 px → NDC mapping done at draw time via uniforms or
            //    a viewport-set scissor rect.
            todo!("wire WGSL + pipeline + quad — pattern matches Plan 03-03 CellPipeline. Aim for < 80 LOC.")
        }

        /// Sets the tint color (sRGB f32 [0,1] per channel + alpha). None disables the stripe.
        pub fn set_color(&mut self, queue: &wgpu::Queue, rgba: Option<[f32; 4]>) {
            self.current_color = rgba;
            if let Some(c) = rgba {
                queue.write_buffer(&self.color_uniform, 0, bytemuck::bytes_of(&c));
            }
        }

        /// Returns the quad vertices spanning `[0, content_width] × [0, 28]` in screen-px coords.
        /// Used by tests + the pipeline's vertex buffer initializer.
        pub fn quad_for(content_width_px: u32) -> [[f32; 2]; 6] {
            // CCW triangle strip → 6 vertices for two triangles covering the rect.
            let w = content_width_px as f32;
            let h = 28.0;
            [
                [0.0, 0.0], [w, 0.0], [0.0, h],
                [w, 0.0], [w, h], [0.0, h],
            ]
        }

        /// Draws the stripe into the active render pass. No-op if `current_color is None`.
        pub fn draw<'a>(&'a self, pass: &mut RenderPass<'a>, bind_group: &'a BindGroup) {
            if self.current_color.is_none() { return; }
            pass.set_pipeline(&self.pipeline);
            pass.set_bind_group(0, bind_group, &[]);
            pass.set_vertex_buffer(0, self.quad_vertices.slice(..));
            pass.draw(0..6, 0..1);
        }
    }
    ```

    ```rust
    //! POLISH-07 / D-75 / UI-SPEC §5.3 — Cmd-Shift-P profile picker (narrow, NOT a command palette).

    use fuzzy_matcher::skim::SkimMatcherV2;
    use fuzzy_matcher::FuzzyMatcher;
    use vector_config::Kind;

    #[derive(Debug, Clone)]
    pub struct PickerEntry {
        pub name: String,
        pub kind: Kind,
    }

    /// D-75: fuzzy-rank profile names. Returns ranked entries (highest score first).
    pub fn match_profiles<'a>(entries: &'a [PickerEntry], query: &str) -> Vec<&'a PickerEntry> {
        let matcher = SkimMatcherV2::default();
        let mut scored: Vec<(i64, &PickerEntry)> = entries.iter()
            .filter_map(|e| matcher.fuzzy_match(&e.name, query).map(|s| (s, e)))
            .collect();
        scored.sort_unstable_by(|a, b| b.0.cmp(&a.0));
        scored.into_iter().map(|(_, e)| e).collect()
    }

    pub struct ProfilePicker {
        pub entries: Vec<PickerEntry>,
        pub query: String,
        pub filtered: Vec<usize>,        // indices into `entries`
        pub selected_idx: usize,
        pub open: bool,
    }

    impl ProfilePicker {
        pub fn new(entries: Vec<PickerEntry>) -> Self {
            let n = entries.len();
            Self { entries, query: String::new(), filtered: (0..n).collect(), selected_idx: 0, open: false }
        }

        pub fn open(&mut self) {
            self.open = true;
            self.query.clear();
            self.filtered = (0..self.entries.len()).collect();
            self.selected_idx = 0;
        }

        pub fn close(&mut self) { self.open = false; }

        pub fn set_query(&mut self, q: &str) {
            self.query = q.to_owned();
            if q.is_empty() {
                self.filtered = (0..self.entries.len()).collect();
            } else {
                let refs: Vec<&PickerEntry> = match_profiles(&self.entries, q);
                self.filtered = refs.into_iter().map(|e| {
                    self.entries.iter().position(|x| x.name == e.name).unwrap()
                }).collect();
            }
            self.selected_idx = 0;
        }

        /// UI-SPEC §5.3: Codespace + DevTunnel rows show `Phase 6+` suffix.
        pub fn row_label(&self, filtered_idx: usize) -> String {
            let entry = &self.entries[self.filtered[filtered_idx]];
            match entry.kind {
                Kind::Local => entry.name.clone(),
                Kind::Codespace | Kind::DevTunnel => format!("{}  Phase 6+", entry.name),
            }
        }

        pub fn select_active(&self) -> Option<&PickerEntry> {
            self.filtered.get(self.selected_idx).map(|&i| &self.entries[i])
        }
    }
    ```

    ```rust
    //! POLISH-01 / POLISH-05 / UI-SPEC §5.4 — toast banner state machine.
    //! Info mode: 36 px, auto-dismiss 5 s. Action mode: 56 px, until-dismissed.

    use std::time::{Duration, Instant};

    #[derive(Debug, Clone, PartialEq, Eq)]
    pub enum ToastMode {
        Info,
        Action { buttons: Vec<String> },
    }

    pub struct ToastBanner {
        pub text: String,
        pub mode: ToastMode,
        pub shown_at: Instant,
        pub focused_button: usize,
    }

    impl ToastBanner {
        pub const INFO_DISMISS_AFTER: Duration = Duration::from_secs(5);

        pub fn info(text: impl Into<String>) -> Self {
            Self { text: text.into(), mode: ToastMode::Info, shown_at: Instant::now(), focused_button: 0 }
        }

        pub fn action(text: impl Into<String>, buttons: Vec<String>) -> Self {
            Self { text: text.into(), mode: ToastMode::Action { buttons }, shown_at: Instant::now(), focused_button: 0 }
        }

        pub fn should_auto_dismiss(&self, now: Instant) -> bool {
            matches!(self.mode, ToastMode::Info) && now.duration_since(self.shown_at) >= Self::INFO_DISMISS_AFTER
        }

        /// UI-SPEC §5.4: chrome.toast.height.info = 36 px; chrome.toast.height.action = 56 px.
        pub fn height_px(&self) -> u32 {
            match self.mode {
                ToastMode::Info => 36,
                ToastMode::Action { .. } => 56,
            }
        }
    }

    /// At most ONE toast visible. New replaces old.
    #[derive(Default)]
    pub struct ToastStack { current: Option<ToastBanner> }
    impl ToastStack {
        pub fn show(&mut self, t: ToastBanner) { self.current = Some(t); }
        pub fn dismiss(&mut self) { self.current = None; }
        pub fn tick(&mut self, now: Instant) {
            if let Some(t) = &self.current {
                if t.should_auto_dismiss(now) { self.current = None; }
            }
        }
        pub fn current(&self) -> Option<&ToastBanner> { self.current.as_ref() }
    }
    ```

    ```rust
    //! POLISH-05 / D-70 — clipboard router.
    //! Consumes ClipboardEvent::Store from Plan 05-05's ForwardingListener; per-origin policy
    //! decides write-through-to-NSPasteboard vs prompt-via-toast.

    use crate::toast::{ToastBanner, ToastStack};
    use vector_config::ClipboardPolicy;
    use vector_term::listener::ClipboardEvent;

    pub struct ClipboardRouter {
        pub active_profile: String,
        pub policy: Option<ClipboardPolicy>,    // resolved per-profile from config; None = ask
    }

    /// Caller-side action requested by the router.
    #[derive(Debug, Clone)]
    pub enum ClipboardAction {
        WritePasteboard(String),
        ShowPrompt(ToastBanner),
        DenyRead,
    }

    impl ClipboardRouter {
        pub fn handle(&self, event: ClipboardEvent, foreground_process: &str) -> ClipboardAction {
            match event {
                ClipboardEvent::Store(_kind, data) => match self.policy {
                    Some(ClipboardPolicy::Allow) => {
                        // base64-decode if needed; alacritty Handler delivers the encoded form.
                        let decoded = base64::engine::general_purpose::STANDARD
                            .decode(&data)
                            .ok()

                        vec!["allow once".to_owned(), "always".to_owned(), "block".to_owned()],
                    )),
                },
                ClipboardEvent::LoadDenied => ClipboardAction::DenyRead,
            }
        }
    }
    ```
