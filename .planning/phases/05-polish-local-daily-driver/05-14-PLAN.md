   /// MEDIUM-1 (05-REVIEWS.md): set NSWindowTabbingModeDisallowed so AppKit will
   /// NEVER group this window into another window's tab bar. Equivalent in
   /// determinism to the C call `[window setTabbingMode:NSWindowTabbingModeDisallowed]`.
   ///
   /// Modeled on the existing `apply_tabbing_identifier` helper (preferred mode).
   /// SAFETY: must be called on the main thread; caller verifies via MainThreadMarker
   /// upstream. The NSWindow pointer is owned by winit's Window for its lifetime.
   pub(crate) unsafe fn apply_tabbing_mode_disallowed(window: &winit::window::Window) {
       use objc2::msg_send;
       use objc2::runtime::AnyObject;
       use raw_window_handle::{HasWindowHandle, RawWindowHandle};

       let Ok(handle) = window.window_handle() else { return; };
       let RawWindowHandle::AppKit(h) = handle.as_raw() else { return; };
       let ns_view: *mut AnyObject = h.ns_view.as_ptr() as *mut AnyObject;
       let ns_window: *mut AnyObject = msg_send![ns_view, window];
       if ns_window.is_null() { return; }
       // NSWindowTabbingModeDisallowed = 2
       let mode: isize = 2;
       let _: () = msg_send![ns_window, setTabbingMode: mode];
   }
