    #[link(name = "Carbon", kind = "framework")]

        fn EnableSecureEventInput();

        pub static DISABLE_COUNT: AtomicUsize = AtomicUsize::new(0);
    }

    #[cfg(test)]
    fn _enable_impl() {
        test_hooks::ENABLE_COUNT.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
    }
    #[cfg(test)]
    fn _disable_impl() {
        test_hooks::DISABLE_COUNT.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
    }
    #[cfg(not(test))]
    fn _enable_impl() { unsafe { EnableSecureEventInput(); } }
    #[cfg(not(test))]
    fn _disable_impl() { unsafe { DisableSecureEventInput(); } }

    pub struct SecureInputGuard {
        enabled: bool,
    }

    impl SecureInputGuard {
        pub fn new() -> Self { Self { enabled: false } }

        pub fn enable(&mut self) {
            if !self.enabled {
                _enable_impl();
                self.enabled = true;
            }
        }

        pub fn disable(&mut self) {
            if self.enabled {
                _disable_impl();
                self.enabled = false;
            }
        }

        pub fn is_enabled(&self) -> bool { self.enabled }

        /// Toggle convenience: returns the new state.
        pub fn toggle(&mut self) -> bool {
            if self.enabled { self.disable(); } else { self.enable(); }
            self.enabled
        }
    }

    impl Drop for SecureInputGuard {
        /// Pitfall 6: ALWAYS disable on drop. If we orphan the secure-event state, OTHER apps
        /// lose keyboard input until the user logs out.
        fn drop(&mut self) {
            self.disable();
        }

    //! 5-selector minimum (D-81): setMarkedText, insertText, unmarkText, hasMarkedText,
    //! markedRange. NO candidate window in v1 (full IME is v2 TERM-V2-01).

    use objc2_foundation::NSRange;
    use tokio::sync::mpsc;

    /// Pure-Rust IME state — testable without AppKit.
    pub struct ImeState {
        preedit: String,
        selected_offset: usize,
        active: bool,
        pub write_tx: mpsc::Sender<Vec<u8>>,
    }

    impl ImeState {
        pub fn new(write_tx: mpsc::Sender<Vec<u8>>) -> Self {
            Self { preedit: String::new(), selected_offset: 0, active: false, write_tx }
        }

        /// setMarkedText: preedit display. **NEVER** writes to PTY (Pitfall 9).
        pub fn set_preedit(&mut self, text: &str, selected_offset: usize) {
            self.preedit = text.to_owned();
            self.selected_offset = selected_offset;
            self.active = !text.is_empty();
        }

        /// insertText: commit. Writes to PTY.
        pub fn commit(&mut self, text: &str) -> bool {
            let sent = self.write_tx.try_send(text.as_bytes().to_vec()).is_ok();
            self.preedit.clear();
            self.active = false;
            sent
        }

        /// unmarkText: drop preedit without committing.
        pub fn clear(&mut self) {
            self.preedit.clear();
            self.active = false;
        }

        pub fn is_active(&self) -> bool { self.active }
        pub fn preedit(&self) -> &str { &self.preedit }
        pub fn marked_range(&self) -> NSRange {
            if self.active {
                NSRange { location: 0, length: self.preedit.chars().count() }
            } else {
                NSRange { location: usize::MAX /* NSNotFound */, length: 0 }
            }
        }
    }
