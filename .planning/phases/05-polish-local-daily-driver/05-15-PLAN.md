Per winit's `Ime` enum: `Preedit { text, cursor_range }` -> `ime.set_preedit(text, selected_offset)`; `Commit(text)` -> `ime.commit(text)`; `Disabled` -> `ime.clear()`.    #[cfg(all(target_os = "macos", not(feature = "test-hooks")))]
   mod appkit_impl {
       use super::ImeState;
       use objc2::{declare_class, msg_send, msg_send_id, mutability, ClassType, DeclaredClass};
       use objc2::rc::Retained;

       declare_class!(

           unsafe impl ClassType for VectorInputView {
               type Super = NSView;
               type Mutability = mutability::InteriorMutable;
               const NAME: &'static str = "VectorInputView";
           }

           impl DeclaredClass for VectorInputView {
               type Ivars = Mutex<ImeState>;
           }

           unsafe impl VectorInputView {
               // 1. insertText:replacementRange:  — D-81 commit path.
               #[method(insertText:replacementRange:)]
               unsafe fn insertText(&self, text: &AnyObject, _range: NSRange) {
                   // text may be NSString or NSAttributedString; coerce by sending -string.
                   let ns_str: Option<Retained<NSString>> = msg_send_id![text, string];
                   let s = ns_str
                       .map(|s| s.to_string())
                       .unwrap_or_else(|| {
                           // fall back: try to cast directly to NSString
                           let direct: Option<&NSString> = (text as *const AnyObject as *const NSString).as_ref();
                           direct.map(|s| s.to_string()).unwrap_or_default()
                       });
                   self.ivars().lock().commit(&s);
               }

               // 2. setMarkedText:selectedRange:replacementRange:  — D-81 preedit path.
               #[method(setMarkedText:selectedRange:replacementRange:)]
               unsafe fn setMarkedText(&self, text: &AnyObject, selected: NSRange, _replacement: NSRange) {
                   let ns_str: Option<Retained<NSString>> = msg_send_id![text, string];
                   let s = ns_str.map(|s| s.to_string()).unwrap_or_default();
                   self.ivars().lock().set_preedit(&s, selected.location);
               }

               // 3. unmarkText
               #[method(unmarkText)]
               unsafe fn unmarkText(&self) {
                   self.ivars().lock().clear();
               }

               // 4. markedRange
               #[method(markedRange)]
               unsafe fn markedRange(&self) -> NSRange {
                   self.ivars().lock().marked_range()
               }

               // 5. selectedRange — return NSNotFound to indicate "no document selection model"
               #[method(selectedRange)]
               unsafe fn selectedRange(&self) -> NSRange {
                   NSRange { location: usize::MAX, length: 0 }
               }

               // hasMarkedText is conventionally also implemented; return YES when active.
               #[method(hasMarkedText)]
               unsafe fn hasMarkedText(&self) -> bool {
                   self.ivars().lock().is_active()
               }
           }
       );

       impl VectorInputView {
           pub fn new_with_state(state: ImeState) -> Retained<Self> {
               let this = Self::alloc().set_ivars(Mutex::new(state));
               unsafe { msg_send_id![super(this), init] }
           }
       }
   }

   #[cfg(all(target_os = "macos", not(feature = "test-hooks")))]
   pub use appkit_impl::VectorInputView;
