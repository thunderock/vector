    ```rust
    //! POLISH-05 / D-71: OSC 52 outbound emitter with tmux-passthrough-safe 58-byte chunking.
    //!
    //! Pitfall 5: tmux 3.4's allow-passthrough writes the inner sequence to the host PTY
    //! in single `write()` calls capped at ~60 chars. Anything longer is truncated.
    //! D-71 fixes the chunk size at 58 (2-byte safety margin).
    //!
    //! The chunking shape — one OSC 52 envelope per chunk, separated by `ESC \ ESC P ESC ]52;c;`
    //! — matches the canonical tmux passthrough recipe (tmux issue #4377).
    //!
    //! D-71 explicitly states: "Vector never re-wraps outbound — that's tmux's job."
    //! We emit RAW OSC 52 (no DCS wrap). The chunk-separator pattern uses `ESC \ ESC P ESC ]52;c;`
    //! purely as a tmux-passthrough resume hint; if no tmux is in the chain, this still
    //! decodes correctly because each chunk is itself a valid OSC 52.

    use base64::{engine::general_purpose::STANDARD, Engine as _};

    const TMUX_CHUNK_MAX: usize = 58;

    pub fn osc52_outbound(payload: &[u8]) -> Vec<u8> {
        let b64 = STANDARD.encode(payload);
        let bytes = b64.as_bytes();
        let mut out = Vec::with_capacity(bytes.len() + 64);

        if bytes.len() <= TMUX_CHUNK_MAX {
            // Single envelope — no chunking needed.
            out.extend_from_slice(b"\x1b]52;c;");
            out.extend_from_slice(bytes);
            out.push(0x07); // BEL
            return out;
        }

        // Multi-chunk: each chunk in its own OSC 52 envelope, separators preserve tmux passthrough.
        let mut first = true;
        for chunk in bytes.chunks(TMUX_CHUNK_MAX) {
            if !first {
                // Close previous chunk's OSC + DCS resume:
                // (the previous chunk already terminated with BEL; we re-open here.)
                out.extend_from_slice(b"\x1b]52;c;");
            } else {
                out.extend_from_slice(b"\x1b]52;c;");
                first = false;
            }
            out.extend_from_slice(chunk);
            out.push(0x07); // BEL
        }
        out
    }

    /// Maximum base64-chars per OSC envelope. Exposed for tests.
    pub const MAX_CHUNK_BASE64: usize = TMUX_CHUNK_MAX;
    ```

    NOTE on the chunking shape: the 05-RESEARCH.md §Example also shows a `\x1b\\` + `\x1bP\x1b]52;c;` separator pattern. Either approach is acceptable so long as (a) each emitted base64 chunk is ≤ 58 bytes AND (b) the cumulative output round-trips through tmux passthrough. The simpler "one OSC 52 envelope per chunk, no DCS wrap" form above honors D-71 ("never re-wrap outbound") most literally. Use this shape.

    ```rust
        let v = std::process::Command::new("tmux").arg("-V").output()

        let ver = String::from_utf8_lossy(&v.stdout);
        // tmux version output: "tmux 3.4" — extract major.minor.
    ```
